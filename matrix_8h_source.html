382  bool compute_inverse(matrix_type& result) const {
383  FT val=FT(0.0), val2=FT(0.0);
384  matrix_type tmp = (*this);
385 
386  result.load_identity();
387 
388  for(index_t i = 0; i != DIM; i++) {
389  val = tmp(i, i); /* find pivot */
390  index_t ind = i;
391  for(index_t j = i + 1; j != DIM; j++) {
392  if(fabs(tmp(j, i)) > fabs(val)) {
393  ind = j;
394  val = tmp(j, i);
395  }
396  }
397 
398  if(ind != i) {
399  for(index_t j = 0; j != DIM; j++) {
400  val2 = result(i, j);
401  result(i, j) = result(ind, j);
402  result(ind, j) = val2; /* swap columns */
403  val2 = tmp(i, j);
404  tmp(i, j) = tmp(ind, j);
405  tmp(ind, j) = val2;
406  }
407  }
408 
409  if(val == 0.0) {
410  return false;
411  }
412 
413  for(index_t j = 0; j != DIM; j++) {
414  tmp(i, j) /= val;
415  result(i, j) /= val;
416  }
417 
418  for(index_t j = 0; j != DIM; j++) {
419  if(j == i) {
420  continue; /* eliminate column */
421  }
422  val = tmp(j, i);
423  for(index_t k = 0; k != DIM; k++) {
424  tmp(j, k) -= tmp(i, k) * val;
425  result(j, k) -= result(i, k) * val;
426  }
427  }
428  }
429 
430  return true;
431  }
